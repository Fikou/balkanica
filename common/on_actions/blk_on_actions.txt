on_startup = {
 events = {
  bosnia_flavor.1
  hum_flavor.1
 }
}

on_monarch_death = {
 events = {
  montenegro_flavor.2
 }
}

on_extend_regency = {
 if = {
  limit = {
   is_subject_of_type = regency
   NOT = { has_country_flag = no_extension_popup }
  }
  if = {
   limit = {
    has_country_flag = forced_regency_extension
   }
   overlord = {
    country_event = {
     id = reg.2
    }
    reverse_add_opinion = {
     who = ROOT
     modifier = force_extended_regency
    }
   }
  }
  else = {
   overlord = {
    country_event = {
     id = reg.1
    }
    add_opinion = {
     who = ROOT
     modifier = peace_extended_regency
    }
   }
   add_country_modifier = {
    name = recently_extended_regency
    duration = 765
    hidden = yes
   }
  }
 }
}

on_heir_needed_theocracy = {
	if = {
  limit = { has_government_attribute = random_kinsman_becomes_heir }
  define_heir = {
   dynasty = ROOT
   age = 15
  }
 }
}
